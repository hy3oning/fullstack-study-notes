-- =========================================  
-- 1. 회원 기본 정보 테이블  
-- =========================================
CREATE TABLE MEMBER (  
  
USER_NO NUMBER(10) NOT NULL, -- 회원 PK  
USER_ID VARCHAR2(50) NOT NULL, -- 로그인 아이디  
USER_PW VARCHAR2(100) NOT NULL, -- 암호화된 비밀번호

EMAIL VARCHAR2(100) NOT NULL, -- 이메일

REG_DATE DATE DEFAULT SYSDATE, -- 가입일  
UPD_DATE DATE DEFAULT SYSDATE, -- 수정일  
  
CONSTRAINT PK_MEMBER PRIMARY KEY (USER_NO),  
CONSTRAINT UK_MEMBER_ID UNIQUE (USER_ID),  
CONSTRAINT UK_MEMBER_EMAIL UNIQUE (EMAIL),
CONSTRAINT UK_MEMBER_PHONE UNIQUE (PHONE),  
CONSTRAINT CK_MEMBER_ENABLED CHECK (ENABLED IN ('0','1'))  
  
);
-- =========================================  
-- 시퀀스
-- =========================================
CREATE SEQUENCE SEQ_MEMBER  
START WITH 1  
INCREMENT BY 1  
NOCACHE;
  
-- 테이블/컬럼 주석  
COMMENT ON TABLE MEMBER IS '회원 기본 정보 테이블';  
COMMENT ON COLUMN MEMBER.USER_NO IS '회원 번호(PK)';  
COMMENT ON COLUMN MEMBER.USER_ID IS '로그인 아이디';  
COMMENT ON COLUMN MEMBER.USER_PW IS '암호화된 비밀번호';  
COMMENT ON COLUMN MEMBER.USER_NAME IS '회원 이름';  
COMMENT ON COLUMN MEMBER.EMAIL IS '이메일';  
COMMENT ON COLUMN MEMBER.PHONE IS '연락처';  
COMMENT ON COLUMN MEMBER.ENABLED IS '계정 활성화 여부 (1: 사용, 0: 정지)';  
COMMENT ON COLUMN MEMBER.REG_DATE IS '가입일';  
COMMENT ON COLUMN MEMBER.UPD_DATE IS '수정일';

ALTER TABLE MEMBER_AUTH ADD CONSTRAINT FK_MEMBER_AUTH_USER 
FOREIGN KEY (USER_NO) REFERENCES MEMBER(USER_NO); 
-- 권한 종류 제한  
ALTER TABLE MEMBER_AUTH ADD CONSTRAINT CK_MEMBER_AUTH 
CHECK (AUTH IN ('ROLE_USER','ROLE_HOST','ROLE_ADMIN'));
 
COMMENT ON TABLE MEMBER_AUTH IS '회원 권한 테이블';
